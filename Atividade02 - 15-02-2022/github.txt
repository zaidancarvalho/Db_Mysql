git init -- iniciar

git branch -M master main -- para trocar do master para o main

git remote add origin -- usar o código do repositorio do github

git remote-v -- para verificar se conectou com o servidor local com o remoto (opcional);

git pull origin main -- puxar os arquivos do repositório para a máquina local (só se tiver o read.me)

git add . -- preparando os arquivos que está na minha pasta.

git status (opcional)

git commit -m "" -- Mensagem para o repositorio do que estou subindo (obrigatorio)

git push -u origin main
